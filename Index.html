public class Calculo_M_P2 {

public static void main(String[] args) {

int Area;

int Perimetro;

double Peri2, Peri1, Peri3;

double PeriTOT;

int X1 ;

int Y1 ;

int X2 ;

int Y2 ;

int X3;

int Y3 ;

int Res ;

do{

JOptionPane.showMessageDialog(null, "VIvenido al programa");

X1 = Integer.parseInt(JOptionPane.showInputDialog("Indique el valor de x1"));

Y1 = Integer.parseInt(JOptionPane.showInputDialog("Indique el valor de y1"));

X2 = Integer.parseInt(JOptionPane.showInputDialog("Indique el valor de x2"));

Y2 = Integer.parseInt(JOptionPane.showInputDialog("Indique el valor de y2"));

X3 = Integer.parseInt(JOptionPane.showInputDialog("Indique el valor de x3"));

Y3 = Integer.parseInt(JOptionPane.showInputDialog("Indique el valor de y3"));

Perimetro = (int) ((int) Math.pow(X2 - X1,2) + Math.pow(Y2 - Y1,2));

Peri1 = Math.sqrt(Perimetro);

JOptionPane.showMessageDialog(null, " La distancia entre el punto A Y el punto B es de " + Peri1);

Perimetro = (int) ((int) Math.pow(X3 - X1,2) + Math.pow(Y3 - Y1,2));

Peri2 = Math.sqrt(Perimetro);

JOptionPane.showMessageDialog(null, " La distancia entre el punto A Y el punto C es de " + Peri2);

Perimetro = (int) ((int) Math.pow(X2 - X3,2) + Math.pow(Y2 - Y3,2));

Peri3 = Math.sqrt(Perimetro);

JOptionPane.showMessageDialog(null, " La distancia entre el punto B Y el punto C es de " + Peri3);

PeriTOT = Peri1 + Peri2 + Peri3 ;

JOptionPane.showMessageDialog(null, "El perimetro de la figura es "+PeriTOT);

Area = Math.abs(X1 * (Y2 - Y3) + X2 * (Y3 - Y1) + X3 * (Y1 - Y2));

JOptionPane.showMessageDialog(null, "El area total del triangulo es\t" + Area );

if (Peri1==Peri2&Peri2==Peri3){ //Equilatero

JOptionPane.showMessageDialog(null, "El triangulo es un Equilatero");

}

if (Peri1==Peri2&Peri2!=Peri3){ //Isoceles

JOptionPane.showMessageDialog(null, "El triangulo es un Isoceles");

}

if (Peri2==Peri3&Peri1!=Peri3){ //Isoceles

JOptionPane.showMessageDialog(null, "El triangulo es un Isoceles");

}

if (Peri2==Peri3&Peri1!=Peri2){ //Isoceles

JOptionPane.showMessageDialog(null, "El triangulo es un Isoceles");

}

if (Peri1!=Peri3&Peri1!=Peri2){ //Escaleno

JOptionPane.showMessageDialog(null, "El triangulo es un Isoceles");

}

Res = JOptionPane.showConfirmDialog(null, "Quiere volver a ejecutar el programa");

}while (Res == 0);

}

}
